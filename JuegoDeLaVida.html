El Juego de La Vida

Reglas:
  1. Si la casilla tiene slo un vecino vivo, entonces la casilla del siguiente renglon (en esa misma columna) vive.
  2. Si la casilla no tiene vecinos vivos, es decir, tiene 2 vecinos muertos, entonces la casilla del siguiente renglon (en esa misma columna) muere.
  3. Si la casilla tiene 2 vecinos vivos, entonces la casilla del siguiente renglon (en esa misma columna) muere.

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.colors import ListedColormap

# Funcin que recibe un tamao de tablero y vector de celdas con condiciones iniciales y devuelve un grfico simulando el juego de la vida
def gameOfLife(n_columnas, n_renglones, celulas, condicion):
    conteo = [np.sum(celulas[0])] # Lista para almacenar el conteo de clulas vivas en cada rengln
    for i in range(1, n_renglones):
        for j in range(1, n_columnas - 1):
            vecinos = [celulas[i-1, j-1], celulas[i-1, j+1]]
            if any(v == 1 for v in vecinos) and any(v == 0 for v in vecinos):
                celulas[i,j] = 1
        conteo.append(np.sum(celulas[i])) # Contamos las clulas vivas y las agregamos a la lista de conteos

    # Generamos la grfica de nuestro juego        
    plt.figure(figsize=(10, 15))  
    plt.imshow(celulas, interpolation='nearest', cmap=ListedColormap(['linen', 'deeppink']))  # Colormap personalizado para representar "vivo" y "muerto"
    plt.xlabel('Casilla', fontstyle='italic')
    plt.ylabel('Tiempo', fontstyle='italic')
    plt.title('Juego de la Vida (Condicin %s)' %condicion, fontweight='bold')
    plt.colorbar(ticks=[0, 1]).set_ticklabels(['Muerto', 'Vivo'])
    
    # Generamos la grfica del conteo de clulas vivas en cada rengln
    plt.figure(figsize=(15, 8))
    plt.plot(range(0, n_renglones), conteo, marker='o', linestyle='-', color='limegreen', label='Clulas Vivas')
    plt.xlabel('Tiempo', fontstyle='italic')
    plt.ylabel('Clulas Vivas', fontstyle='italic')
    plt.title('Conteo de Clulas Vivas a lo largo del Tiempo (Condicin %s)' %condicion, fontweight='bold')
    plt.grid(True, linestyle='--', alpha=0.6)
    plt.legend(loc='best')

    plt.show()
